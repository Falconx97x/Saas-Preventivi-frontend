"use client";

import { useState, useEffect } from "react";

interface Client {
  id: number;
  name: string;
  email: string;
}

interface QuoteItem {
  description: string;
  quantity: number;
  price: number;
}

export default function CreateQuotePage() {
  const [clients, setClients] = useState<Client[]>([]);
  const [selectedClient, setSelectedClient] = useState<number | null>(null);
  const [title, setTitle] = useState("");
  const [items, setItems] = useState<QuoteItem[]>([
    { description: "", quantity: 1, price: 0 },
  ]);

  // Carica clienti
  useEffect(() => {
    const fetchClients = async () => {
      try {
        const res = await fetch("http://localhost:3001/clients");
        if (!res.ok) throw new Error("Errore caricamento clienti");
        const data: Client[] = await res.json();
        setClients(data);
        if (data.length > 0) setSelectedClient(data[0].id);
      } catch (error) {
        console.error(error);
      }
    };
    fetchClients();
  }, []);

  // Aggiorna singolo item
  const updateItem = (index: number, field: keyof QuoteItem, value: string | number) => {
    setItems((prev) =>
      prev.map((item, i) =>
        i === index
          ? {
              ...item,
              [field]:
                field === "quantity" || field === "price"
                  ? Number(value)
                  : value,
            }
          : item
      )
    );
  };

  const addItem = () => setItems([...items, { description: "", quantity: 1, price: 0 }]);
  const removeItem = (index: number) => setItems(items.filter((_, i) => i !== index));

  const total = items.reduce((sum, item) => sum + item.quantity * item.price, 0);

  // Salva preventivo
  const saveQuote = async () => {
    if (!title || !selectedClient) {
      alert("Inserisci titolo e seleziona cliente");
      return;
    }

    const cleanItems = items.map((i) => ({
      description: i.description,
      quantity: Number(i.quantity),
      price: Number(i.price),
    }));

    try {
      const res = await fetch("http://localhost:3001/quotes", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          title,
          clientId: selectedClient,
          items: cleanItems,
        }),
      });

      if (!res.ok) {
        const data = await res.json();
        throw new Error(data.error || "Errore creazione preventivo");
      }

      alert("Preventivo salvato!");
      setTitle("");
      setItems([{ description: "", quantity: 1, price: 0 }]);
    } catch (error: any) {
      console.error(error);
      alert(error.message);
    }
  };

  return (
    <div className="p-8 text-black">
      <h1 className="text-3xl font-bold mb-4">Crea Preventivo</h1>

      {/* Selezione cliente */}
      <div className="mb-4">
        <label className="block mb-1 font-semibold">Cliente</label>
        <select
          className="border p-2 rounded w-full"
          value={selectedClient ?? ""}
          onChange={(e) => setSelectedClient(Number(e.target.value))}
        >
          {clients.map((c) => (
            <option key={c.id} value={c.id}>
              {c.name} ({c.email})
            </option>
          ))}
        </select>
      </div>

      {/* Titolo preventivo */}
      <div className="mb-4">
        <label className="block mb-1 font-semibold">Titolo preventivo</label>
        <input
          type="text"
          className="border p-2 rounded w-full"
          value={title}
          onChange={(e) => setTitle(e.target.value)}
        />
      </div>

      {/* Lista articoli */}
      <div className="mb-4">
        <h2 className="font-semibold mb-2">Articoli</h2>
        {items.map((item, index) => (
          <div key={index} className="flex gap-2 mb-2">
            <input
              type="text"
              placeholder="Descrizione"
              className="border p-2 flex-1 rounded"
              value={item.description}
              onChange={(e) => updateItem(index, "description", e.target.value)}
            />
            <input
              type="number"
              placeholder="Quantità"
              className="border p-2 w-24 rounded"
              value={item.quantity}
              onChange={(e) => updateItem(index, "quantity", e.target.value)}
            />
            <input
              type="number"
              placeholder="Prezzo"
              className="border p-2 w-24 rounded"
              value={item.price}
              onChange={(e) => updateItem(index, "price", e.target.value)}
            />
            <button
              onClick={() => removeItem(index)}
              className="bg-red-500 text-white px-2 rounded"
            >
              X
            </button>
          </div>
        ))}
        <button
          onClick={addItem}
          className="bg-green-500 text-white px-4 py-2 rounded mt-2"
        >
          Aggiungi Articolo
        </button>
      </div>

      <h2 className="mt-4 font-bold">Totale: €{total}</h2>

      <button
        onClick={saveQuote}
        className="bg-blue-500 text-white px-6 py-2 rounded font-semibold"
      >
        Salva Preventivo
      </button>
    </div>
  );
}
